** Provide function for fixing interaction between (minor) mode and evil
   #+begin_src elisp
     (defun my/keymap-fix-with-evil-mode (mode &optional hook map)
       "Allows (minor) mode to use evil as a base, but its keybindings take precedence.

     Solution derived from: https://github.com/emacs-evil/evil/issues/511
     "
       (let* ((base (symbol-name mode))
	      (hook (if hook hook (intern (concat base "-hook"))))
	      (map (if map map (intern (concat base "-map")))))

	 (evil-set-initial-state mode 'normal)
	 (eval-after-load mode
	   '(progn
	      ;; module's keybindings should have stronger priority than
	      ;; normal-mode keybindings:
	      (evil-make-overriding-map map 'normal)
	      ;; force evil keymaps to update after module loaded:
	      (add-hook hook #'evil-normalize-keymaps)))))
   #+end_src

     (use-package evil-surround
       :preface
       ;; Source: https://github.com/emacs-evil/evil-surround
       (defmacro define-and-bind-quoted-text-object (name key start-regex end-regex)
         "Create new evil text objects so that we can change/delete
     the appropriate delimiters."
         (let ((inner-name (make-symbol (concat "evil-inner-" name)))
               (outer-name (make-symbol (concat "evil-a-" name))))
           `(progn
              (evil-define-text-object ,inner-name (count &optional beg end type)
                (evil-select-paren ,start-regex ,end-regex beg end type count nil))
              (evil-define-text-object ,outer-name (count &optional beg end type)
                (evil-select-paren ,start-regex ,end-regex beg end type count t))
              (define-key evil-inner-text-objects-map ,key #',inner-name)
              (define-key evil-outer-text-objects-map ,key #',outer-name))))
       :after evil
       :config
       ;; Handle org-mode markup delimiters:
       (define-and-bind-quoted-text-object "pipe" "|" "|" "|")
       (define-and-bind-quoted-text-object "equals" "=" "=" "=")
       (define-and-bind-quoted-text-object "slash" "/" "/" "/")
       (define-and-bind-quoted-text-object "plus" "+" "+" "+")
       (define-and-bind-quoted-text-object "asterisk" "*" "*" "*")
       ;; sometimes your have to escape the regex
       (define-and-bind-quoted-text-object "dollar" "$" "\\$" "\\$")
       (global-evil-surround-mode 1))

      (use-package better-narrowing
        :no-require t
        :straight nil
        :preface
        (defun my/narrow-or-widen-dwim (p)
	  "If the buffer is narrowed, it widens. Otherwise, it narrows intelligently.
      Intelligently means: region, org-src-block, org-subtree, or defun,
      whichever applies first.
      Narrowing to org-src-block actually calls `org-edit-src-code'.

      With prefix P, don't widen, just narrow even if buffer is already
      narrowed."
	  (interactive "P")
	  (declare (interactive-only))
	  (cond ((and (buffer-narrowed-p) (not p)) (widen))
	        ((region-active-p)
		 (narrow-to-region (region-beginning) (region-end)))
	        ((derived-mode-p 'org-mode)
		 ;; `org-edit-src-code' is not a real narrowing command.
		 ;; Remove this first conditional if you don't want it.
		 (cond ((ignore-errors (org-edit-src-code))
		        (delete-other-windows))
		       ((org-at-block-p)
		        (org-narrow-to-block))
		       (t (org-narrow-to-subtree))))
	        (t (narrow-to-defun))))
        :general
        (leader "n" '(my/narrow-or-widen-dwim :which-key "narrow/widen")))

   #+begin_src elisp
     (defun slugify (title)
       (let ((slug-trim-chars
              '(;; Combining Diacritical Marks https://www.unicode.org/charts/PDF/U0300.pdf
                768 ; U+0300 COMBINING GRAVE ACCENT
                769 ; U+0301 COMBINING ACUTE ACCENT
                770 ; U+0302 COMBINING CIRCUMFLEX ACCENT
                771 ; U+0303 COMBINING TILDE
                772 ; U+0304 COMBINING MACRON
                774 ; U+0306 COMBINING BREVE
                775 ; U+0307 COMBINING DOT ABOVE
                776 ; U+0308 COMBINING DIAERESIS
                777 ; U+0309 COMBINING HOOK ABOVE
                778 ; U+030A COMBINING RING ABOVE
                779 ; U+030B COMBINING DOUBLE ACUTE ACCENT
                780 ; U+030C COMBINING CARON
                795 ; U+031B COMBINING HORN
                803 ; U+0323 COMBINING DOT BELOW
                804 ; U+0324 COMBINING DIAERESIS BELOW
                805 ; U+0325 COMBINING RING BELOW
                807 ; U+0327 COMBINING CEDILLA
                813 ; U+032D COMBINING CIRCUMFLEX ACCENT BELOW
                814 ; U+032E COMBINING BREVE BELOW
                816 ; U+0330 COMBINING TILDE BELOW
                817 ; U+0331 COMBINING MACRON BELOW
                )))
         (cl-flet* ((nonspacing-mark-p (char) (memq char slug-trim-chars))
                    (strip-nonspacing-marks (s) (string-glyph-compose
                                                 (apply #'string
                                                        (seq-remove #'nonspacing-mark-p
                                                                    (string-glyph-decompose s)))))
                    (cl-replace (title pair) (replace-regexp-in-string (car pair) (cdr pair) title)))
           (let* ((pairs `(("[^[:alnum:][:digit:]]" . "_") ;; convert anything not alphanumeric
                           ("__*" . "_")                   ;; remove sequential underscores
                           ("^_" . "")                     ;; remove starting underscore
                           ("_$" . "")))                   ;; remove ending underscore
                  (slug (-reduce-from #'cl-replace (strip-nonspacing-marks title) pairs)))
             (downcase slug)))))
   #+end_src
