* <<<treemacs>>> ([[https://github.com/Alexander-Miller/treemacs][github]]): file drawer
  #+begin_src elisp
    (use-package treemacs
      :preface
      (defvar treemacs-ignored-regexes nil
        "A list of regexes that Treemacs should not display.")

      (defun my/inject-dir-locals (&rest _)
        (message "Injecting dir-locals.el")
        (hack-dir-local-variables-non-file-buffer))

      (defun my/treemacs-root ()
        (when-let ((buffer (treemacs-get-local-buffer)))
          (with-current-buffer buffer
            (let* ((btn (treemacs-current-button))
                   (parent (treemacs-button-get btn :parent)))
              (while parent
                (setq btn parent)
                (setq parent (treemacs-button-get btn :parent)))
              (treemacs-button-get btn :path)))))

      :hook ((after-init . treemacs-hide-gitignored-files-mode)
             (treemacs-follow-mode . my/inject-dir-locals)
             (treemacs-project-follow-mode . my/inject-dir-locals)
             (treemacs-select . my/inject-dir-locals)
             (treemacs-pre-refresh . my/inject-dir-locals))
      :custom
      (treemacs-width 30)
      ;; (treemacs-tag-follow-delay 0.5)
      :config
      ;; (treemacs-project-follow-mode)

      ;; slightly lower the size of treemacs icons
      (treemacs-resize-icons 16)

      ;; Doesn't work when set via :custom
      (setq treemacs-collapse-dirs 0)

      (treemacs-git-mode 'deferred)

      ;; (treemacs-tag-follow-mode t)
      (treemacs-filewatch-mode t)
      (treemacs-fringe-indicator-mode 'always)

      (defun my/treemacs-ignored (absolute-path git-info)
        "Filters any buttons whose paths match regexes in`treemacs-ignored-regexes'."
        (-some (lambda (re) (s-matches? re absolute-path)) treemacs-ignored-regexes))

      (dolist (f '(treemacs-is-file-git-ignored? my/treemacs-ignored))
        (add-to-list 'treemacs-pre-file-insert-predicates f))

      (require 'treemacs-project-follow-mode)
      (defun my/treemacs-jump-to-buffer ()
        "Navigate treemacs to root of current buffer.

    This will also:
    - inject any local dir-locals.el variables into the treemacs buffer, and
    - trigger a `treemacs-refresh'."
        (interactive)
        (treemacs--do-follow-project)
        (with-selected-window (treemacs-get-local-window)
          (setq treemacs-ignored-regexes nil
                file-local-variables-alist nil
                dir-local-variables-alist nil)
          (hack-dir-local-variables-non-file-buffer)
          (treemacs-refresh)))

      :general
      (local-leader
        "*" '(my/treemacs-jump-to-buffer :which-key "treemacs update"))
      (leader
        "fd"  '(treemacs :which-key "file-drawer")
        "fD"  '(treemacs-bookmark :which-key "file-drawer (from bookmark)")))
  #+end_src
