* <<<treemacs>>> ([[https://github.com/Alexander-Miller/treemacs][github]]): file drawer
  #+begin_src elisp
    (use-package treemacs
      :custom
      (treemacs-width 30)
      (treemacs-tag-follow-delay 0.5)
      (treemacs-hide-gitignored-files-mode t)
      :config
      ;; slightly lower the size of treemacs icons
      (treemacs-resize-icons 18)

      (treemacs-tag-follow-mode t)
      (treemacs-filewatch-mode t)
      (treemacs-fringe-indicator-mode 'always)

      (pcase (cons (not (null (executable-find "git")))
                   (not (null treemacs-python-executable)))
        (`(t . t)
         (treemacs-git-mode 'deferred))
        (`(t . _)
         (treemacs-git-mode 'simple)))

      (define-inline my/treemacs-ignore-pycache (file _)
        (declare (side-effect-free t) (pure t))
        (inline-letevals (file)
          (inline-quote
           (string-equal ,file "__pycache__"))))
      (push 'my/treemacs-ignore-pycache treemacs-ignored-file-predicates)

      (defun my/treemacs-ignored (filename absolute-path)
        (or
         (s-equals? filename "latex-images")
         (s-equals? filename "ltximg")
         (s-contains? ".lsp" filename)
         (s-contains? ".clj-kondo" filename)))
      (push 'my/treemacs-ignored treemacs-ignored-file-predicates)

      ;; (treemacs-display-current-project-exclusively)
      :general
      (leader
        "fd"  '(treemacs :which-key "file-drawer")
        "fD"  '(treemacs-bookmark :which-key "file-drawer (from bookmark)")))
  #+end_src
