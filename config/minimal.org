This should ideally be the absolute bare minimum required to bootstrap an emacs
you can work with.

* Enable lexical binding
#+begin_src elisp
  ;; -*- lexical-binding: t -*-
#+end_src

* Use [[https://github.com/hrs/sensible-defaults.el][sensible-defaults]]
  #+begin_src elisp
    (load-file (concat user-emacs-directory "/config/" "sensible-defaults.el"))
    (sensible-defaults/use-all-settings)
    (sensible-defaults/use-all-keybindings)
    (sensible-defaults/backup-to-temp-directory)
  #+end_src

* No littering, please
  #+begin_src elisp
    (use-package no-littering
      :config
      (require 'no-littering)
      (setq custom-file (no-littering-expand-etc-file-name "custom.el")))
  #+end_src

* Favour unix line-endings
  #+begin_src elisp
    (set-buffer-file-coding-system 'unix 't)
  #+end_src

* OSX-specific options
  Right option-key should be pass-through (not meta), to allow typing ~#~, for
  example:
  #+begin_src elisp
    (setq mac-right-option-modifier 'none)
  #+end_src

    (use-package general
      :preface
      (defun kill-other-buffers ()
        "Kill all other buffers."
        (interactive)
        (mapc 'kill-buffer
              (delq (current-buffer)
                    (org-remove-if-not 'buffer-file-name (buffer-list)))))

      (defun my/emacs-lisp-files ()
        (-remove (lambda (s) (or (s-ends-with? ".dir-locals.el" s)
                            (s-ends-with? ".dirs-local.el" s))) ; general.el
                 (apply 'append
                        (-map (lambda (dir) (directory-files dir t "\.el$" t)) load-path))))

      (defun my/find-emacs-lisp-file ()
        "Prompt the user to choose an Emacs Lisp file from the load-path."
        (interactive)
        (let* ((files (my/emacs-lisp-files))
               (file-names (mapcar (-compose #'f-no-ext #'f-filename) files))
               (alist (cl-mapcar #'cons files file-names))
               (prompt "Choose an Emacs Lisp file: ")
               (vertico-sort-function #'vertico-sort-alpha)
               (chosen-file (completing-read prompt file-names)))
          (find-file (car (rassoc chosen-file alist)))))

      (defun my/eshell ()
        "Switch to eshell buffer if it exists; otherwise, activate it."
        (interactive)
        (if (and (boundp 'eshell-buffer-name)
                 (get-buffer-window eshell-buffer-name 'visible))
             (switch-to-buffer eshell-buffer-name)
           (eshell)))

      :after which-key
