This should ideally be the absolute bare minimum required to bootstrap an emacs
you can work with.

* Enable lexical binding
#+begin_src emacs-lisp
  ;; -*- lexical-binding: t -*-
#+end_src

* Use [[https://github.com/hrs/sensible-defaults.el][sensible-defaults]]
  #+begin_src emacs-lisp
    (load-file (concat user-emacs-directory "/config/" "sensible-defaults.el"))
    (sensible-defaults/use-all-settings)
    (sensible-defaults/use-all-keybindings)
    (sensible-defaults/backup-to-temp-directory)
  #+end_src

* No littering, please
  #+begin_src emacs-lisp
    (use-package no-littering
      :config
      (require 'no-littering)
      (setq custom-file (no-littering-expand-etc-file-name "custom.el")))
  #+end_src

* Favour unix line-endings
  #+begin_src emacs-lisp
    (set-buffer-file-coding-system 'unix 't)
  #+end_src

* OSX-specific options
  Right option-key should be pass-through (not meta), to allow typing ~#~, for
  example:
  #+begin_src emacs-lisp
    (setq mac-right-option-modifier 'none)
  #+end_src

* <<<which-key>>> ([[https://github.com/justbur/emacs-which-key][github]]): Display leader/prefix hints
  #+begin_src emacs-lisp
    (use-package which-key
      :config
      ;; (which-key-setup-side-window-right-bottom)
      (which-key-mode 1)
      :custom
      (which-key-sort-order 'which-key-key-order-alpha)
      (which-key-idle-delay 0.4)
      ;; (which-key-show-operator-state-maps t)
      (which-key-allow-evil-operators t))
  #+end_src

* <<<general>>> ([[https://github.com/noctuid/general.el][github]]): Evil key binding helper
  #+begin_src emacs-lisp
    (use-package general
      :preface
      (defun kill-other-buffers ()
	"Kill all other buffers."
	(interactive)
	(mapc 'kill-buffer
	      (delq (current-buffer)
		    (org-remove-if-not 'buffer-file-name (buffer-list)))))
      :after which-key
      :config
      (general-override-mode 1)

      (general-create-definer leader
	:states '(normal visual insert motion emacs)
	:keymaps 'override
	:prefix "SPC"
	:non-normal-prefix "C-SPC")

      (general-create-definer local-leader
	:states '(normal visual insert motion emacs)
	:keymaps 'override
	:prefix ","
	:non-normal-prefix "C-,")

      (general-define-key
       :keymaps 'key-translation-map
       "ESC" (kbd "C-g"))

      (winner-mode)
      (leader
	""     nil
	;; "c"   (general-simulate-key "C-c" :which-key "C-c")
	"c"   '(org-capture :which-key "capture")
	":"   'eval-expression
	"SPC" '(execute-extended-command :which-key "M-x")

* <<<evil>>> ([[https://github.com/emacs-evil/evil][github]]): Vim-style keybindings
  #+begin_src emacs-lisp
    (use-package evil
      :demand
      :init
      ;; Necessary for evil-collection
      (setq evil-want-keybinding nil)
      :custom
      (evil-undo-system 'undo-redo)
      :config
      ;; For some reason this needs to be set here:
      (customize-set-variable 'evil-want-Y-yank-to-eol t)
      (evil-mode 1))
  #+end_src
