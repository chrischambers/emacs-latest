*** ansible-doc ([[https://github.com/emacsorphanage/ansible-doc][github]])
    #+begin_src elisp
      (use-package ansible-doc
        :preface
        (require 'ansible-doc)

        (defun my/python-variable-at-point ()
          "Extracts word at point conforming to python variable rules."
          (let (start end (pattern "[:alpha:][[:alpha:][:digit:]._="))
            (save-excursion
              (skip-chars-backward pattern)
              (setq start (point))
              (skip-chars-forward pattern)
              (setq end (point))
              (buffer-substring-no-properties start end))))

        (defun my/ansible-doc-read-module (prompt)
          "Read an Ansible module name from minibuffer (including
      fully-qualified ansible modules) with PROMPT."
          (let* ((modules (ansible-doc-modules))
                 (symbol (my/python-variable-at-point))               ; modified
                 (symbol (s-replace "ansible.builtin." "" symbol))    ; modified
                 (default (if (or (null modules) (member symbol modules))
                              symbol
                            nil))
                 (prompt (if default
                             (format "%s (default %s): " prompt default)
                           (format "%s: " prompt)))
                 ;; If we have no modules available, we don't require a match, and use
                 ;; the symbol at point as default value and sole completion candidate.
                 (reply (completing-read prompt
                                         (or modules (list default))
                                         nil (not (null modules))
                                         nil nil default)))
            (if (string= reply "") default reply)))

        (advice-add 'ansible-doc-read-module :override #'my/ansible-doc-read-module)
        :general
        ('(motion normal) yaml-mode-map "K" #'ansible-doc))
    #+end_src
