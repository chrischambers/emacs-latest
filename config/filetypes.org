* <<<conf-mode>>>: Config files
  #+begin_src emacs-lisp
    (use-package conf-mode
      :mode ("\\.git.*$"
             "dot_.*"
             "\\.rc$"
             "\\.conf$"
             "git.*\\.symlink$"
             "rc\\.symlink$"
             "conf\\.symlink$"))
  #+end_src

* <<<logview>>>: Log files
  #+begin_src emacs-lisp
    (use-package logview
      :commands logview-mode
      :mode
      ("\\.log\\(?:\\.[0-9]+\\)?\\'" . logview-mode)
      ("syslog\\(?:\\.[0-9]+\\)?\\'" . logview-mode))
  #+end_src
