* <<<conf-mode>>>: Config files
  #+begin_src emacs-lisp
    (use-package conf-mode
      :mode ("\\.git.*$"
             "dot_.*"
             "\\.rc$"
             "\\.conf$"
             "git.*\\.symlink$"
             "rc\\.symlink$"
             "conf\\.symlink$"))
  #+end_src

* <<<logview>>>: Log files
  #+begin_src emacs-lisp
    (use-package logview
      :commands logview-mode
      :mode
      ("\\.log\\(?:\\.[0-9]+\\)?\\'" . logview-mode)
      ("syslog\\(?:\\.[0-9]+\\)?\\'" . logview-mode))
  #+end_src

** YAML (esp. for Ansible)
*** COMMENT yaml-mode (superseded by poly-ansible)
   #+begin_src emacs-lisp
     (use-package yaml-mode
       :mode ("\\.ya?ml\\'"))
   #+end_src

*** COMMENT emacs-ansible
    #+begin_src emacs-lisp
      (use-package ansible
        :straight (emacs-ansible
                   :host github
                   :repo "k1LoW/emacs-ansible"
                   ;; The following prevents pulling in the entire ansible codebase:
                   :nonrecursive t))
    #+end_src

*** poly-ansible
    #+begin_src emacs-lisp
      (use-package poly-ansible
        :mode ("\\.ya?ml\\'" . poly-ansible-mode))
    #+end_src

*** ansible-doc
    #+begin_src emacs-lisp
      (use-package ansible-doc)
    #+end_src

*** company-ansible
    #+begin_src emacs-lisp
      (use-package company-ansible
        :config
        (add-company-backend 'company-ansible))
    #+end_src

*** YAML Linting via flymake
    #+begin_src emacs-lisp
      (use-package flymake-yamllint
        :hook ((yaml-mode . flymake-yamllint-setup)
               (yaml-mode . flymake-mode)))
    #+end_src
