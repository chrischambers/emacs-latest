      ;; https://emacs.stackexchange.com/questions/33892/replace-element-of-alist-using-equal-even-if-key-does-not-exist/33893#33893
      (defun alist-set (key val alist &optional symbol)
        "Set property KEY to VAL in ALIST. Return new alist.
            This creates the association if it is missing, and otherwise sets
            the cdr of the first matching association in the list. It does
            not create duplicate associations. By default, key comparison is
            done with `equal'. However, if SYMBOL is non-nil, then `eq' is
            used instead.

            This method may mutate the original alist, but you still need to
            use the return value of this method instead of the original
            alist, to ensure correct results."
        (if-let ((pair (if symbol (assq key alist) (assoc key alist))))
            (setcdr pair val)
          (push (cons key val) alist))
        alist)

      (defun my/open-gif (file link)
        (let ((my-image (create-image file))
              (tmpbuf (get-buffer-create "*gif")))
          (switch-to-buffer tmpbuf)
          (erase-buffer)
          (insert-image my-image)
          (image-animate my-image nil t)
          ))

      (customize-set-variable 'org-file-apps
                              (alist-set "\\.gif\\'" #'my/open-gif org-file-apps))

      (use-package org-capture
        :init
        (setq my/org-capture-templates (concat org-directory "/" "templates/"))
        (setq my/default-template (concat my/org-capture-templates "default.org"))
        :preface
        (defun my/get-filepath-from-title (&optional subdirectory)
          "Prompt for a title, return filepath located under org-directory.

      If `subdirectory' is specified, that will be included in the path.

      SIDE-EFFECT: Creates ancestor directories for `file-path' if they do not exist."
          (let* ((p (or subdirectory "/"))
                 (p (if (not (s-starts-with? "/" p)) (s-prepend "/" p) p))
                 (p (if (not (s-ends-with? "/" p))   (s-append  "/" p) p))
                 (path (concat org-directory p))
                 (title (read-string "Title: "))
                 (slug (slugify title))
                 (file-path (concat path slug ".org")))
            (make-directory path t)
            file-path))
        :straight nil
        :custom
        (org-default-notes-file (concat org-directory "/todo.org"))
        (org-capture-templates
         `(("t" "Task" entry
            (file+headline "todo.org" "Inbox")
            "* TODO %?")
           ("w" "Word" plain
            (file (lambda () (my/get-filepath-from-title "memex/#words")))
            (file ,(concat my/org-capture-templates "word.org")))

   #+begin_src emacs-lisp
     (with-eval-after-load 'org
     (require 'f)
     (require 's)
     (require 'dash)

     (defcustom org-agenda-blacklist '("#old" "#archives/completed_tasks.org$")
       "patterns to exclude from org-agenda path names")

     (defvar org-todo-states (-map (lambda (s) (car (s-split "(" s)))
                                   (seq-drop (nth 0 org-todo-keywords) 1))
       "org-todo-keywords without the suffixes")

     (defvar org-agenda-todo-regex
       (let* ((prepend-star (lambda (entry) (concat "\\* " entry)))
              (interpolate-|-char (lambda (cur acc) (concat acc "\\|" cur)))
              (headlines (-map prepend-star org-todo-states))
              (pattern (-reduce interpolate-|-char headlines)))
         pattern)
       "Regex which filters all TODO keywords")

     (defun org-agenda-find-files-with (regex)
       "Yields a fresh array with all files containing todos which match REGEX.

     Uses grep to discover all files containing anything stored in org-agenda-todo-regex."
       (-reject #'file-directory-p
                (split-string
                 (shell-command-to-string
                  (concat "grep --include=\"*.org\" -rl -e '" regex (concat "' " org-directory)))
                 "\n")))

     (defun s-matches-any? (regexes s)
       "True if any of the patterns match s"
       (pcase regexes
         (`(,head . ,tail) (or (s-match head s) (s-matches-any? tail s)))
         (_ nil)))

     (defun my/org-agenda-blacklist-match (string)
       "True when string is a match against any entry in the org-agenda-blacklist"
       (s-matches-any? org-agenda-blacklist string))

     (defun my/org-agenda-sort (files)
       "Sorts agenda files appropriately."
       (let* ((decorated (-map (lambda (f) (cons (f-filename f) f)) files))
              (sorted (sort decorated (lambda (p1 p2) (string< (car p1) (car p2)))))
              (undecorated (-map #'cdr sorted)))
        undecorated))

     (defun my/org-agenda-file-filter (files)
       (let ((files (-reject #'my/org-agenda-blacklist-match files)))
        (pcase-let*
            ((`(,root-files ,files)
              (-separate (lambda (f) (s-match (f-canonical (concat org-directory "/" "[^/]*\.org$")) f)) files))
             (`(,project-files ,files)
              (-separate (lambda (f) (s-match (f-canonical (concat org-directory "/projects/" ".*\.org$")) f)) files)))
          (-flatten (-map #'my/org-agenda-sort (list root-files project-files files))))))

     (setq org-agenda-files
            (my/org-agenda-file-filter
             (org-agenda-find-files-with org-agenda-todo-regex))))
   #+end_src

     (with-eval-after-load 'org-roam
       (defun my/conditional-kill-buffer-for-file (file)
	 "Kill buffer for FILE if it's not modified: otherwise, throw error."
	 (interactive "f")
	 (when-let ((buffer (find-buffer-visiting file)))
	   (if (buffer-modified-p buffer)
	       (error (format "Error: buffer '%s' has not been saved."
			      (buffer-name buffer)))

       (defun my--memex-relocate (old-file filename &optional new-root)
	 (let* ((new-root (concat org-roam-directory "/memex/"))
		(new-file (concat (read-directory-name "> " new-root) filename)))
	   (f-move old-file new-file)
	   (find-file new-file)
	   (vc-refresh-state)
	   (org-roam-db-sync)
	   (message (format "Migrated '%s' to %s." filename new-file))))

       (defun my/promote-node ()
	 "Moves the file for the selected node from the old hierarchy into
     the current memex."
	 (interactive)
	 (let* ((old-root (concat org-roam-directory "/#old/"))
		(matches? (lambda (node)
			    (f-descendant-of? (org-roam-node-file node) old-root)))
		(node (org-roam-node-read "" matches?))
		(old-file (org-roam-node-file node))
		(filename (f-filename old-file)))
	   (my/conditional-kill-buffer-for-file old-file)
	   (my--memex-relocate old-file filename)))

		(new-root (concat org-roam-directory "/memex/"))

		(buffer-path (buffer-file-name buffer)))
