    (use-package gptel
      :preface
      (defun my/anthropic-api-key ()
        (let ((entry (car (auth-source-search
                           :host "api.anthropic.com"
                           :user "apikey"
                           :require '(:secret)))))
          (when entry
            (let ((secret (plist-get entry :secret)))
              (if (functionp secret)
                  (funcall secret)
                secret)))))
