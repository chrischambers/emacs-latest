     (with-eval-after-load 'gptel
       (require 'subr-x)
       (require 'gptel)
       (require 'gptel-request)

       (defcustom gptel-session-automatic-save nil
         "Whether to automatically save new GPTel sessions to disk."
         :type 'boolean)

       (defcustom gptel-session-automatic-save-on-response nil
         "Whether to automatically save after LLM responses."
         :type 'boolean)

       (defcustom gptel-session-dir (expand-file-name "~/llm_sessions/")
         "Directory to save GPTel sessions to."
         :type 'directory)

       (defun gptel-session-filename-date-desc-model (desc &optional date model)
         "Return session filename as 2026-31-01::something-i-asked::chatgpt-gpt-5-2.org

     DESC is slugified. DATE defaults to today (YYYY-MM-DD).
     MODEL defaults to (gptel--model-tag)."
         (let ((desc (gptel--slugify desc 70))
               (date (or date (format-time-string "%Y-%m-%d")))
               (model (or model (gptel--model-tag))))
           (format "%s::%s::%s.org" date desc model)))

       (defun gptel-session-filename-date/desc::model (desc &optional date model)
         "Return session filename as 2026-31-01/something-i-asked::chatgpt-gpt-5-2.org

     DESC is slugified. DATE defaults to today (YYYY-MM-DD).
     MODEL defaults to (gptel--model-tag)."
         (let ((desc (gptel--slugify desc 70))
               (date (or date (format-time-string "%Y-%m-%d")))
               (model (or model (gptel--model-tag))))
           (format "%s/%s::%s.org" date desc model)))

       (defun gptel-session-filename-date/desc/model (desc &optional date model)
         "Return session filename as 2026-31-01/something-i-asked/chatgpt-gpt-5-2.org

     DESC is slugified. DATE defaults to today (YYYY-MM-DD).
     MODEL defaults to (gptel--model-tag)."
         (let ((desc (gptel--slugify desc 70))
               (date (or date (format-time-string "%Y-%m-%d")))
               (model (or model (gptel--model-tag))))
           (format "%s/%s/%s.org" date desc model)))

       (defcustom gptel-session-name-function #'gptel-session-filename-date-desc-model
         "Function used for naming session files."
         :type 'function)

       (defvar-local gptel--session-initialized nil)

       (defun gptel--slugify (s &optional maxlen)
         (let* ((s (downcase (string-trim s)))
                (s (replace-regexp-in-string "[^a-z0-9]+" "-" s))
                (s (replace-regexp-in-string "\\`-+\\|-+\\'" "" s))
                (maxlen (or maxlen 70)))
           (cond
            ((string-empty-p s) "chat")
            ((> (length s) maxlen) (substring s 0 maxlen))
            (t s))))

       (defun gptel--model-tag ()
         (gptel--slugify
          (format "%s-%s" (gptel-backend-name gptel-backend) gptel-model)
          90))

       (defun gptel--buffer-excerpt (&optional maxchars)
         "Excerpt of the buffer to help the LLM infer a title."
         (let ((maxchars (or maxchars 2200)))
           (buffer-substring-no-properties
            (point-min)
            (min (point-max) (+ (point-min) maxchars)))))

       (defun gptel-infer-title-async (excerpt callback)
         "Infer a short conversation title from EXCERPT and call CALLBACK.

     CALLBACK is called as (CALLBACK SUGGESTION), where SUGGESTION is a
     slugified string suitable for use in filenames.

     If inference fails, CALLBACK is called with nil.

     This is asynchronous; it does not block Emacs."
         (declare (indent 1))
         (let* (;; Keep this request cheap and deterministic-ish.
                (system "You generate concise conversation titles.
     Return a short title (2–5 words). No quotes. No punctuation at end. Title case.")
                (prompt (format "Generate a short title for this conversation:\n\n%s" excerpt)))
           (gptel-request
               prompt
             :system system
             :stream nil
             :callback
             (lambda (response _info)
               (condition-case _err
                   (let* ((title (and (stringp response) response))
                          (suggest (gptel--slugify title 70)))
                     (funcall callback suggest))
                 (error (funcall callback nil)))))))

       (defun gptel-session-rename-file (&optional buffer)
         "Infer a title for BUFFER (default: current), prompt user, then rename file."
         (interactive)
         (let* ((buffer (or buffer (current-buffer)))
                (excerpt (with-current-buffer buffer (gptel--buffer-excerpt))))
           (gptel-infer-title-async
               excerpt
             (lambda (suggest)
               (when (buffer-live-p buffer)
                 (with-current-buffer buffer
                   (let* ((suggest (or suggest "chat"))
                          ;; daemon-safe: prompt when Emacs is ready to read minibuffer
                          (final-desc (read-string "gptel session description: " suggest))
                          (new-name (funcall gptel-session-name-function final-desc))
                          (new-path (expand-file-name new-name gptel-session-dir))
                          (parent (f-parent new-path)))
                     (make-directory parent t)
                     (set-visited-file-name new-path t t)
                     (save-buffer))))))))

       (defun gptel-session--init-on-first-send (&rest _)
         "Hook/advice target: run once per GPTel chat buffer."
         (when (and gptel-session-automatic-save
                    (bound-and-true-p gptel-mode)
                    (not gptel--session-initialized)
                    (not (buffer-file-name)))
           (setq gptel--session-initialized t)
           (gptel-session-rename-file (current-buffer))))

       (defun gptel-session--save-buffer-on-response (beg end)
         (when gptel-session-automatic-save-on-response
           (let ((buf-path (or (buffer-file-name (current-buffer)) "")))
             (when (f-ancestor-of? gptel-session-dir buf-path)
               (save-buffer)))))

       (advice-add 'gptel-send :after #'gptel-session--init-on-first-send)
       (add-hook 'gptel-post-response-functions #'gptel-session--save-buffer-on-response))
