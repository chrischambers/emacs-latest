* Dictionaries
** Dictionary Config
   #+begin_src elisp
     (use-package dictionary-config
       :no-require t
       :straight nil
       :general
       (leader
         "d"    '(:ignore t :which-key "dict/spell")
         "ds"   '(ispell-buffer :which-key "spellcheck buffer")
         "dS"   '(ispell-word :which-key "spellcheck at point")
         "df"   '(dictionary-search :which-key "dictionary (find)")
         "dd"   '(dictionary-lookup-definition :which-key "dictionary (point)")))
   #+end_src

* Spell Checking
** COMMENT <<<flyspell>>>: Spell-checker
   #+begin_src elisp
     (use-package flyspell
       :hook
       ((prog-mode . flyspell-prog-mode)
        (text-mode . flyspell-mode)
        (gfm-mode . flyspell-mode)
        (git-commit-mode . flyspell-mode)
        (mu4e-compose-mode . flyspell-mode))
       :custom
       (ispell-dictionary "british")
       (ispell-personal-dictionary (concat user-emacs-directory "dictionary.txt"))
       ;; Apparently a major performance gain:
       (flyspell-issue-message-flag nil))
   #+end_src

     (use-package jinx
       :preface
       :hook (emacs-startup . global-jinx-mode)
       :bind (("M-$" . jinx-correct)
              ("C-M-$" . jinx-languages)
              ([remap ispell-buffer] . jinx-correct-all)
              ([remap ispell-word] . jinx-correct)
              ([remap evil-next-flyspell-error] . jinx-next)
              ([remap evil-prev-flyspell-error] . jinx-previous)))
   #+end_src
