     (use-package repl-toggle
       :preface
       (defun my/cider-set-ns (&rest r)
         (run-with-timer 0.5 nil 'cider-repl-set-ns my/cider-current-ns)
         (remove-hook 'cider-repl-mode-hook #'my/cider-set-ns))

       (defun my/notes-active-within-buffer? ()
         (interactive)
         (s-starts-with? my/notes-directory (buffer-file-name) t))

       (defun my/possibly-launch-project-repl (expanded _params &optional cljs-p)
         "Evaluate EXPANDED code block using cider.
     When CLJS-P is non-nil, use a cljs connection instead of clj.
     The PARAMS from Babel are not used in this function."
         (org-require-package 'cider "Cider")
         (let* ((connection (cider-current-connection (if cljs-p "cljs" "clj"))))
           (unless connection
             (if (my/notes-active-within-buffer?)
                 (cider-jack-in `(:project-dir ,my/notes-directory))
               (sesman-start-session 'CIDER)))
           (if (not connection)
               ;; Display in the result instead of using `user-error'
               "Please reevaluate when nREPL is connected"
             (let ((response (nrepl-sync-request:eval expanded connection)))
               (or (nrepl-dict-get response "root-ex")
                   (nrepl-dict-get response "ex")
                   (nrepl-dict-get response "out"))))))

       (advice-add 'ob-clojure-eval-with-cider :override #'my/possibly-launch-project-repl)

       (defun my/cider-startup ()
         "Jack in, set current ns and provide a handy cider usage tip as well."
         (setq my/cider-current-ns (cider-current-ns))
         (add-hook 'cider-repl-mode-hook #'my/cider-set-ns)
         (let ((params (if (my/notes-active-within-buffer?)
                           `(:project-dir ,my/notes-directory)
                         nil)))
           (and (cider-jack-in params)
                (cider-drink-a-sip))))

      ;; Source: https://docs.projectile.mx/projectile/projects.html#test-dirsrc-dir-vs-related-files-fn

      (with-eval-after-load 'projectile
        (defun my-get-python-test-file (impl-file-path)
          "Return the corresponding test file directory for IMPL-FILE-PATH"
          (let* ((rel-path (f-relative impl-file-path (projectile-project-root)))
                 (src-dir (car (f-split rel-path))))
            (cond ((f-exists? (f-join (projectile-project-root) "test"))
                   (projectile-complementary-dir impl-file-path src-dir "test"))
                  ((f-exists? (f-join (projectile-project-root) "tests"))

                  (t (error "Could not locate a test file for %s!" impl-file-path)))))

          "Return the corresponding impl file directory for TEST-FILE-PATH"
          (if-let* ((root (projectile-project-root))
                    (rel-path (f-relative test-file-path root))
                    (src-dir-guesses `(,(f-base root) ,(downcase (f-base root)) "src"))
                    (src-dir (cl-find-if (lambda (d) (f-exists-p (f-join root d)))
                                         src-dir-guesses)))
              (projectile-complementary-dir test-file-path "tests?" src-dir)
            (error "Could not locate a impl file for %s!" test-file-path)))

        (projectile-update-project-type
         'python-pkg
         :src-dir #'my-get-python-impl-dir
         :test-dir #'my-get-python-test-dir))

                                                     (region-beginning)
