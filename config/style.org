** Custom Theme Selector
   [[https://emacs.stackexchange.com/questions/24088/make-a-function-to-toggle-themes][Make a function to toggle themes - Emacs Stack Exchange]]

   #+begin_src elisp
     (use-package theme-cycler
       :no-require t
       :straight nil
       :demand t
       :preface
       (defcustom my/themes (custom-available-themes) "The themes to cycle through")
       (defcustom my/themes-index 0 "Initial index of my/themes")
       :custom
       (my/themes '(doom-dracula modus-operandi poet nano))
       :config
       (defun my/select-theme (f)
         (interactive)
         (let ((themes-length (length my/themes))
               (index (funcall f my/themes-index)))
           (setq my/themes-index (% (+ index themes-length) themes-length))
           (my/load-indexed-theme)))

       (defun my/next-theme ()
         "Moves to next theme in list."
         (interactive)
         (my/select-theme '1+))

       (defun my/previous-theme ()
         "Moves to previous theme in list."
         (interactive)
         (my/select-theme '1-))

       (defun my/load-indexed-theme ()
         (let ((selected-theme (nth my/themes-index my/themes)))
           (message (concat "loading theme: " (symbol-name selected-theme)))
           (load-theme selected-theme t)))

       (load-theme (nth my/themes-index my/themes) t)
       :general
       (leader
         "T"  '(:ignore t :which-key "theme")
         "Tc"  '(load-theme :which-key "choose")
         "Tp" '(my/previous-theme :which-key "previous")
         "Tn" '(my/next-theme :which-key "next")))
   #+end_src
