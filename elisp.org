* Emacs-lisp bindings
  #+begin_src emacs-lisp
    (use-package my/emacs-lisp-config
      :no-require t
      :straight nil
      :general
      ('motion emacs-lisp-mode-map "K" 'helpful-at-point)
      (local-leader emacs-lisp-mode-map
        "e"  '(:ignore t :major-modes emacs-lisp-mode :which-key "eval")
        "ee" '(eval-last-sexp :which-key "expression")
        "ef" '(eval-defun :which-key "function")
        "eF" '(load-file :which-key "file")
        "er" '(eval-region :which-key "region")
        "eb" '(eval-buffer :which-key "buffer")))
  #+end_src

* Improving Emacs-lisp

** Add =backward-line= function, for consistency
  #+begin_src emacs-lisp
    (defun backward-line (&optional N)
      "Move N lines backwards (forward if N is negative)."
      (interactive)
      (let ((n (if N N 1)))
        (forward-line (* -1 n))))
  #+end_src
