* Bootstrap config-related functions
   #+begin_src emacs-lisp
     (setq config-file (expand-file-name
                        (concat user-emacs-directory "configuration.org")))
   #+end_src

   #+begin_src emacs-lisp
    (defun config-edit ()
    (interactive)
    (find-file config-file))
   #+end_src

   #+begin_src emacs-lisp
    (defun config-reload ()
       (interactive)
       (org-babel-load-file config-file))
   #+end_src

 #+begin_src emacs-lisp
   (defun my/config-load (name)
     "Used for loading other org config files from `configuration.org'"
     (interactive)
     (org-babel-load-file (expand-file-name name user-emacs-directory)))
 #+end_src
