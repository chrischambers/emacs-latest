** Function for Opening in Browser
   #+begin_src elisp
     (defun my/org-src-path ()
       (if (not (org-src-edit-buffer-p))
           (message "Not in an Org src edit buffer")
         (let* ((src-buffer (org-src-source-buffer)) ;; Get main Org buffer
                (pos (with-current-buffer src-buffer
                       (org-babel-where-is-src-block-head)))) ;; Get block start position
           (if (not pos)
               (message "Could not locate source block")
             (with-current-buffer src-buffer
               (goto-char pos) ;; Move to source block
               (let* ((element (org-element-at-point))
                      (file-name (buffer-file-name (current-buffer)))
                      (dir (f-parent file-name))
                      (params (org-element-property :parameters element))
                      (tangle (alist-get :tangle (org-babel-parse-header-arguments params))))
                 (if tangle (f-join dir tangle) file-name)))))))

     (defun my/open-path-in-browser (path)
       (let* ((base-cmd  "open -a /Applications/Firefox\\ Developer\\ Edition.app")
              (cmd (s-join " " (list base-cmd path))))
         (message cmd)
         (shell-command cmd)))

     (defun my/open-in-browser (buffer)
       (interactive "b")
       (with-current-buffer buffer
         (let ((path (if (org-src-edit-buffer-p)
                         (my/org-src-path)
                       (buffer-file-name (current-buffer)))))
           (when path
             (my/open-path-in-browser path)))))

     (defun my/get-tangle-property ()
       (cdr (assq :tangle (org-babel-parse-header-arguments
                           (org-element-property :parameters
                                                 (org-element-at-point))))))

     (defun my/tangle-then-browse ()
       (interactive)
       (when-let ((file-name (my/get-tangle-property))
                  (parent-path (f-parent (buffer-file-name (current-buffer))))
                  (path (f-join parent-path file-name)))
         (org-babel-tangle)
         (my/open-path-in-browser path)))
   #+end_src

    (defun eval-clojure-doc-current-line ()
      "Evaluate current line with clojure.repl/doc and insert result below."
      (interactive)
      (when (and (eq major-mode 'org-mode)
                 (featurep 'cider)
                 (cider-connected-p))
        (let* ((current-line (string-trim (thing-at-point 'line t)))
               (doc-expr (format "(with-out-str (clojure.repl/doc %s))" current-line))
               (result (nrepl-dict-get (cider-nrepl-sync-request:eval doc-expr) "value")))
          (end-of-line)
          (newline)
          (insert (read result)))))

** Migrate Clojure Code to Notes
   #+begin_src elisp

     (defvar my/clojure-project-root
       (f-join my/notes-directory "code/clojure")
       "Root of the Clojure code within notes")

     (defun my/migrate-clojure-code-to-notes (buffer-name)
       (interactive "b")
       (when (member (f-ext buffer-name) (list "clj" "cljc"))
         (let* ((buffer (get-buffer buffer-name))
                (source1 (buffer-file-name buffer))

                (projectile-create-missing-test-files nil)
                (source2 (projectile-find-implementation-or-test source1))

                (prefix (f-common-parent (list source1 source2)))
                (dest1 (->> source1
                            (s-chop-prefix prefix)
                            (f-join my/clojure-project-root)))
                (dest2 (->> source2
                            (s-chop-prefix prefix)
                            (f-join my/clojure-project-root))))
           ;; (message "%s" (list dest1 dest2))
           (when (not (f-exists? dest1))
             (message "Copying from %s to %s..." source1 dest1)
             (f-copy source1 dest1))
           (when (and (f-exists? source2) (not (f-exists? dest2)))
             (message "Copying from %s to %s..." source2 dest2)
             (f-copy source2 dest2))
           (progn
             (kill-buffer buffer)
             (find-file dest1)))))

   #+begin_src elisp
     (defun my/org-build-latex-previews (dir)
       "Rebuild LaTeX previews in all .org files under DIR.

     Note: this paradigm of reusing a temp buffer, looping over the files,
     inserting their contents, processing and suppressing the save flag is
     much faster than simply looping over the files, opening, processing and
     killing them.
     "
       (require 'org)

       ;; Required in batch
       (setq org-preview-latex-default-process 'dvisvgm)
       (setq org-startup-with-latex-preview nil)

       (let ((files (directory-files-recursively dir "\\.org$"))
             (count 0))
         (dolist (file files)
           (with-temp-buffer
             (let ((kill-buffer-query-functions nil))
               (insert-file-contents file)
               (setq buffer-file-name file)
               (org-mode)

               ;; Skip if no LaTeX fragments anywhere in file
               (goto-char (point-min))
               (org-clear-latex-preview)
               (org-latex-preview '(16))
               (setq count (1+ count))

               ;; Silence modification state
               (set-buffer-modified-p nil))))
         count))

     (defun my/org-build-latex-previews-subprocess (dir)
       "Launch `~/bin/emacs-generate-latex-previews' helper."
       (interactive "DDirectory: ")
       (let ((process-environment
              (cons "TERM=xterm-256color" process-environment)))
         (start-process
          "emacs-generate-latex-previews"
          "*emacs-generate-latex-previews*"
          "emacs-generate-latex-previews" dir)))

     (defun my/async-org-build-latex-previews (dir)
       "This doesn't work correctly, as `org--latex-preview-region' requires a
     graphical interface, but the async process is headless.

     I'm keeping it around for the useful patterns - i.e. pushing onto the
     log, injecting variables.
     "
       (interactive "DDirectory: ")
       (async-start
        `(lambda ()
           (let ((log)
                 (org-files (directory-files-recursively ,dir "\\.org$")))
             ,(async-inject-variables "org-latex")
             ,(async-inject-variables "org-preview")
             (push (format "latex: %s" (executable-find "latex")) log)
             (push (format "dvipng: %s" (executable-find "dvipng")) log)
             (push (format "org-preview-latex-default-process: %s" org-preview-latex-default-process) log)
             (dolist (file org-files)
               (push (format "Processing %s" file) log)
               (with-current-buffer (find-file-noselect file)
                 ;; Disable confirmation prompts for LaTeX previews
                 (let ((org-startup-with-latex-preview nil))
                   (org-mode) ;; Ensure org-mode is enabled
                   (org-clear-latex-preview (point-min) (point-max))
                   (org--latex-preview-region (point-min) (point-max))
                   (when (buffer-modified-p)
                     (push (format "Writing buffer %s" (buffer-name)) log)
                     (save-buffer)))
                 ;; Close the buffer without saving if you opened it just for preview generation
                 (kill-buffer)))
             (nreverse log)))
        (lambda (log)
          (progn
            (message "Latex rebuild finished asynchronously.")
            (inspect log)))))

    (defcustom my/cider-debug-local-exclude-regexes '("^argspec" "^retspec" "^guardrails-wrapper")
      "Patterns to exclude from cider-debug-locals"
      :type '(list regexp))

    (defun my/blacklisted-local? (word)
      (s-matches-any? my/cider-debug-local-exclude-regexes word))

    (defun my/cider-debug-filter-locals (f locals)
      (funcall f (-remove
                  (-compose #'my/blacklisted-local? #'-first-item)
                  (sort locals :key #'-first-item ))))

                :around
                #'my/cider-debug-filter-locals)
