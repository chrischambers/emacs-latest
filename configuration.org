* Function for Opening in Browser
  #+begin_src elisp
    (defun my/org-src-path ()
      (if (not (org-src-edit-buffer-p))
          (message "Not in an Org src edit buffer")
        (let* ((src-buffer (org-src-source-buffer)) ;; Get main Org buffer
               (pos (with-current-buffer src-buffer
                      (org-babel-where-is-src-block-head)))) ;; Get block start position
          (if (not pos)
              (message "Could not locate source block")
            (with-current-buffer src-buffer
              (goto-char pos) ;; Move to source block
              (let* ((element (org-element-at-point))
                     (file-name (buffer-file-name (current-buffer)))
                     (dir (f-parent file-name))
                     (params (org-element-property :parameters element))
                     (tangle (alist-get :tangle (org-babel-parse-header-arguments params))))
                (if tangle (f-join dir tangle) file-name)))))))

    (defun my/open-path-in-browser (path)
      (let* ((base-cmd  "open -a /Applications/Firefox\\ Developer\\ Edition.app")
             (cmd (s-join " " (list base-cmd path))))
        (message cmd)
        (shell-command cmd)))

    (defun my/open-in-browser (buffer)
      (interactive "b")
      (with-current-buffer buffer
        (let ((path (if (org-src-edit-buffer-p)
                        (my/org-src-path)
                      (buffer-file-name (current-buffer)))))
          (when path
            (my/open-path-in-browser path)))))

    (defun my/get-tangle-property ()
      (cdr (assq :tangle (org-babel-parse-header-arguments
                          (org-element-property :parameters
                                                (org-element-at-point))))))

    (defun my/tangle-then-browse ()
      (interactive)
      (when-let ((file-name (my/get-tangle-property))
                 (parent-path (f-parent (buffer-file-name (current-buffer))))
                 (path (f-join parent-path file-name)))
        (org-babel-tangle)
        (my/open-path-in-browser path)))
  #+end_src

* Messing around with nix-shell integration
  #+begin_src elisp
    (require 'subr-x)  ;; For string-trim

    (defvar my/nix-package-cache nil
      "Cache of available Nix packages for completion.")

    (defvar my/original-environment nil
      "Stores the original environment variables before applying Nix environment.")

    (defun my/get-available-nix-packages ()
      "Retrieve a list of available Nix packages and cache them."
      (or my/nix-package-cache
          (let* ((cmd "nix search nixpkgs/nixpkgs-unstable ^ --json 2> /dev/null | jq 'keys[]'")
                 (output (shell-command-to-string cmd))
                 (sanitised (replace-regexp-in-string "\"[^\\.]+\\.[^\\.]+\\.\\([^\"]+\\)\"" "\\1" output))
                 (result (split-string sanitised "\n" t)))
            (setq my/nix-package-cache result))))

    (defun my/capture-shell-environment ()
      "Capture the original environment variables before applying Nix environment."
      (when (not my/original-environment)
      (setq my/original-environment (copy-alist process-environment))))

    (defun my/restore-shell-environment ()
      "Restore the original environment variables."
      (interactive)
      (when my/original-environment
        (setq process-environment (copy-alist my/original-environment))
        (setq exec-path (split-string (getenv "PATH") path-separator))
        (setq my/original-environment nil)
        (message "Original environment restored successfully!")))

    (defun my/update-emacs-env-from-nix (packages)
      "Update Emacs environment variables using `nix-shell -p` with PACKAGES."
      (interactive
       (list (completing-read-multiple
              "Select Nix packages (comma-separated): "
              (my/get-available-nix-packages))))
      (my/capture-shell-environment)
      (let* ((package-args (mapconcat 'identity packages " "))
             (cmd (format "nix-shell -p %s --run 'env'" package-args))
             (output (shell-command-to-string cmd)))
        (message cmd)
        (message output)
        (dolist (line (split-string output "\n" t))
          (when (string-match "^\\([^=]+\\)=\\(.*\\)" line)
            (let ((var (match-string 1 line))
                  (value (match-string 2 line)))
              (setenv var value)
              (when (string= var "PATH")
                (setq exec-path (split-string value path-separator))))))
        (message "Emacs environment updated successfully with selected Nix packages!")))

    (defun my/clojure-remove-top-level-requires (&optional buffer)
      (interactive "b")
      (let ((buffer (or buffer (current-buffer))))
        (with-current-buffer buffer
          (replace-regexp "^(require [^)]*)\n*" ""))))

    (defun my/clojure-replace-ns (replacement)
      (save-excursion
        (goto-char (point-min))
        (when (re-search-forward "^(ns " nil t)
          (re-search-backward "(")
          (let ((start (point))
                (end (progn (forward-sexp) (point))))
            (replace-region-contents start end (lambda () replacement))
            (my/clojure-remove-top-level-requires)
            (save-buffer)))))

    (defun my/clojure-consolidate-ns (&rest libspecs)
      (let* ((file-name (buffer-file-name (current-buffer)))
             (libspecs (-map (lambda (ls) (format "'%s'" ls)) libspecs))
             (cmd (concat "clj_consolidate_requires " file-name " " (s-join " " libspecs)))
             (new-ns (s-trim (shell-command-to-string cmd))))
        (when new-ns
          (my/clojure-replace-ns new-ns))))
